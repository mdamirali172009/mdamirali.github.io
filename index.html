import sys from PyQt6.QtWidgets import (QApplication, QMainWindow, QWidget, QVBoxLayout, QHBoxLayout, QPushButton, QLabel, QFileDialog, QSlider, QStyle) from PyQt6.QtCore import Qt from moviepy.editor import VideoFileClip class PremiumVideoEditor(QMainWindow): def __init__(self): super().__init__() self.setWindowTitle("ProEdit Elite - Next Gen Editor") self.resize(1000, 700) # প্রিমিয়াম ডার্ক থিম স্টাইলশিট self.setStyleSheet(""" QMainWindow { background-color: #121212; } QPushButton { background-color: #1DB954; color: white; border-radius: 8px; padding: 10px; font-weight: bold; font-size: 14px; } QPushButton:hover { background-color: #1ed760; } QLabel { color: #E0E0E0; font-family: 'Segoe UI'; } #PreviewArea { background-color: #000000; border: 2px solid #333; border-radius: 10px; } QSlider::handle:horizontal { background: #1DB954; width: 18px; border-radius: 9px; } """) self.initUI() def initUI(self): main_layout = QVBoxLayout() # --- উপরের অংশ: প্রিভিউ উইন্ডো --- self.preview_label = QLabel("ভিডিও প্রিভিউ এখানে দেখা যাবে") self.preview_label.setAlignment(Qt.AlignmentFlag.AlignCenter) self.preview_label.setObjectName("PreviewArea") self.preview_label.setMinimumHeight(400) main_layout.addWidget(self.preview_label) # --- মাঝের অংশ: কন্ট্রোলস --- controls = QHBoxLayout() self.import_btn = QPushButton("Import Video") self.import_btn.clicked.connect(self.import_video) self.play_btn = QPushButton() self.play_btn.setIcon(self.style().standardIcon(QStyle.StandardPixmap.SP_MediaPlay)) self.export_btn = QPushButton("Export 4K") self.export_btn.setStyleSheet("background-color: #6200EE;") controls.addWidget(self.import_btn) controls.addWidget(self.play_btn) controls.addWidget(self.export_btn) main_layout.addLayout(controls) # --- নিচের অংশ: টাইমলাইন স্লাইডার --- timeline_label = QLabel("Timeline / Seek Bar") main_layout.addWidget(timeline_label) self.slider = QSlider(Qt.Orientation.Horizontal) main_layout.addWidget(self.slider) central_widget = QWidget() central_widget.setLayout(main_layout) self.setCentralWidget(central_widget) def import_video(self): file_path, _ = QFileDialog.getOpenFileName(self, "ভিডিও ফাইল সিলেক্ট করুন", "", "Video Files (*.mp4 *.avi *.mkv)") if file_path: self.preview_label.setText(f"Loaded: {file_path.split('/')[-1]}") # এখানে ভিডিও লোড করার লজিক আসবে video = VideoFileClip(file_path) print(f"ভিডিও ডিউরেশন: {video.duration} seconds") if __name__ == "__main__": app = QApplication(sys.argv) editor = PremiumVideoEditor() editor.show() sys.exit(app.exec())
